              <script>
                this.on('mount', function() {
                  // URL変更を検知開始
                  riot.route.start()
                  // routerに変更があったら実行される
                  riot.route(function(tag) {
                    // barタグにtagの内容を展開する
                    riot.mount('bar', tag)
                  })
                  // リロード
                  riot.route(location.hash.replace('#', ''))
                })
              </script>
            </foo>
